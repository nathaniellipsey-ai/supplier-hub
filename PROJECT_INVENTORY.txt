    - Referential integrity
    - ACID transactions
    - Auto-timestamps

Schema:
  suppliers table (5+ records)
    - id (PK), supplier_id (UNIQUE), name
    - email, phone, address, city, state, zip_code, country
    - category, status
    - created_at, updated_at

  products table (10+ records)
    - id (PK), supplier_id (FK)
    - product_code (UNIQUE), product_name
    - description, unit_cost
    - lead_time_days, min_order_qty
    - created_at

  search_history table
    - id (PK), search_query, results_count
    - user_id, created_at

  - idx_product_supplier (product lookup)
